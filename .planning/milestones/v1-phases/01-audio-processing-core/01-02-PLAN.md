    import { NextRequest, NextResponse } from "next/server";
    import { generateUploadToken } from "@/lib/vercel-blob";
    import type { UploadRequest, UploadResponse } from "@/lib/types";
    import { z } from "zod";

    // Disable caching for upload endpoint
    export const dynamic = "force-dynamic";

    // Zod schema for request validation
    const UploadRequestSchema = z.object({
      filename: z.string().min(1, "filename is required"),
      contentType: z.string().optional(),
    });

    export async function POST(request: NextRequest) {
      try {
        const body = await request.json();

        // Validate request body
        const validation = UploadRequestSchema.safeParse(body);
        if (!validation.success) {
          return NextResponse.json(
            { error: validation.error.errors[0].message },
            { status: 400 }
          );
        }

        const { filename } = validation.data;

        // Generate upload token for client-side upload
        const tokenData = await generateUploadToken(filename);

        const response: UploadResponse = {
          uploadUrl: tokenData.uploadUrl,
          token: tokenData.token,
        };

        return NextResponse.json(response, { status: 200 });
      } catch (error) {
        const message = error instanceof Error ? error.message : "Unknown error";
        console.error("Upload token generation failed:", error);

        return NextResponse.json(
          { error: message },
          { status: 503 }
        );
      }
    }
    ```

    import { NextRequest, NextResponse } from "next/server";
    import { submitTranscription } from "@/lib/assemblyai";
    import type { TranscribeRequest, TranscribeResponse } from "@/lib/types";
    import { z } from "zod";

    // Disable caching for transcription endpoint
    export const dynamic = "force-dynamic";

    // Zod schema for request validation
    const TranscribeRequestSchema = z.object({
      audioUrl: z.string().url("audioUrl must be a valid URL"),
    });

    export async function POST(request: NextRequest) {
      try {
        const body = await request.json();

        // Validate request body
        const validation = TranscribeRequestSchema.safeParse(body);
        if (!validation.success) {
          return NextResponse.json(
            { error: validation.error.errors[0].message },
            { status: 400 }
          );
        }

        const { audioUrl } = validation.data;

        // Submit transcription job to AssemblyAI
        const jobId = await submitTranscription(audioUrl);

        const response: TranscribeResponse = {
          jobId,
          status: "queued",
        };

        return NextResponse.json(response, { status: 201 });
      } catch (error) {
        const message = error instanceof Error ? error.message : "Unknown error";
        console.error("Transcription submission failed:", error);

        return NextResponse.json(
          { error: message },
          { status: 500 }
        );
      }
    }
    ```
